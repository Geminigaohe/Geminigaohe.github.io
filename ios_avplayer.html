使用 AVPlayer 播放网络音乐的一个简单的例子:

//
//  ViewController.m
//  MyPlayer
//
//  Created by GaoHe on 16/7/12.
//  Copyright © 2016年 GaoHe. All rights reserved.
//


// 引入 AVFoundation 框架
#import <AVFoundation/AVFoundation.h>
#import "ViewController.h"

@interface ViewController() {
    // 播放列表
    NSArray *_arrayPlayList;
    // 播放器
    AVPlayer *_player;
    // 当前播放的引索值
    int _currentPlayIndex;
    // 播放状态
    BOOL _isPlaying;
}

// 播放进度 Slider
@property (weak, nonatomic)IBOutlet UISlider *sliderProgress;

// 缓存进度
@property (weak, nonatomic) IBOutlet UIProgressView *progressView;

//播放按钮
@property (weak, nonatomic) IBOutlet UIButton *playButton;

@end


@implementation ViewController

- (void)setPlayList {
    // 设置播放列表
    NSURL *playUrl_1 = [NSURL URLWithString:playUrl_1];

    NSURL *playUrl_2 = [NSURL URLWithString:playUrl_2];

    NSURL *playUrl_3 = [NSURL URLWithString:playUrl_3];

    NSURL *playUrl_4 = [NSURL URLWithString:playUrl_4];

    _arrayPlayList = @[playUrl_1, playUrl_2, playUrl_3, playUrl_4];
}

- (void)viewDidLoad {
    [super viewDidLoad];
    self.view.backgroundColor = [UIColor whiteColor];

    // 程序开始时没有播放
    [self.playButton setEnabled:NO];
    self.sliderProgress.value = 0;
    _isPlaying = NO;
    [self setPlayList];

    AVPlayerItem *item = [[AVPlayerItem alloc] initWithURL:playUrl_1];

    // 初始化播放器, 默认播放第一首
    _player = [[AVPlayer alloc] initWithPlayerItem:item];
    // 播放完成后发出通知
    [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(playedidEnd) name:AVPlayerItemDidPlayToEndTimeNotification object:nil];
    // 添加观察者, 用来监视播放器的变化
    [_player addObserver:self forKeyPath:@"status" options:NSKeyValueObservingOptionNew context:nil];

    __weak  ViewController *me = self;
    __weak AVPlayer *weakPlayer = _player;

    [_player addPeriodicTimeObserverForInterval:CMTimeMake(1,1) queue:dispatch_get_main_queue() usingBlock:^(CMTime time) {

        // 获取当前播放的时间
        float currentTime = weakPlayer.currentItem.currentTime.value/weakPlayer.currentItem.currentTime.timescale;

        me.sliderProgress.value = currentTime;

        // 计算缓冲进度
        NSTimeInterval  timeInterval = [me availableDuration];
        NSLog(@"Time Interval:%f", timeInterval);
        CMTime duration = weakPlayer.currentItem.duration;
        CGFloat  totalDuration = CMTimeGetSeconds(duration);

        // 更新缓冲进度
        [me.progressView setProgress:timeInterval/totalDuration animated:YES];
    }];
}

// 计算缓冲进度
- (NSTimeInterval)availableDuration {
    NSArray *loadedTimeRanges = [[_player currentItem] loadedTimeRanges];
    CMTimeRange timeRange = [loadedTimeRanges.firstObject CMTimeRangeValue];
    float startSeconds = CMTimeGetSeconds(timeRange.start);
    float durationSeconds = CMTimeGetSeconds(timeRange.duration);
    NSTimeInterval result = startSeconds + durationSeconds;
    return result;
}

// 播放
- (IBAction)doPlay:(id)sender {
    if (_isPlaying) {
        [_player pause];
        _isPlaying = NO;
        [self.playButton setTitle:@"播放" forState:UIControlStateNormal];
    } else {
        [_player play];
        [self.playButton setTitle:@"暂停" forState:UIControlStateNormal];
        _isPlaying = YES;
        self.sliderProgress.maximumValue = _player.currentItem.duration.value/_player.currentItem.duration.timescale;
    }
}

// 跳到指定时间播放
- (IBAction)doChangeProgress:(id)sender {
    [_player seekToTime:CMTimeMake(self.sliderProgress.value, 1)];
}

// 上一首
- (IBAction)doForward:(id)sender {
    if (_currentPlayIndex-1 >= 0) {
        AVPlayerItem *item = [[AVPlayerItem alloc] initWithURL:_arrayPlayList[_currentPlayIndex-1]];
        [_player replaceCurrentItemWithPlayerItem:item];
        [_player play];
        _currentPlayIndex--;
    }
}

// 下一首
- (IBAction)doNext:(id)sender {
    if (_currentPlayIndex+1 < _arrayPlayList.count) {
        AVPlayerItem *item = [[AVPlayerItem alloc] initWithURL:_arrayPlayList[_currentPlayIndex+1]];
        [_player replaceCurrentItemWithPlayerItem:item];
        [_player play];
        _currentPlayIndex++;
    }
}

// 播放完毕后调用的方法
- (void)playedidEnd {
    NSLog(@"播放完毕");
}

// 观察者方法, 用来监听播放器状态
- (void)observeValueForKeyPath:(NSString *)keyPath ofObject:(id)object change:(NSDictionary *)change context:(void *)context {
    if ([keyPath isEqualToString:@"status"]) {
        if (_player.status == AVPlayerItemStatusReadyToPlay) {
            [self.playButton setEnabled:YES];
        }
    }
}

@end
